FROM publications

10) Show data richness per profile

SELECT 
    p.full_name,
    COUNT(DISTINCT pub.id) AS publications,
    COUNT(DISTINCT d.id) AS degrees
FROM profiles p
LEFT JOIN publications pub ON p.profile_id = pub.profile_id
LEFT JOIN degrees_titles d ON p.profile_id = d.profile_id
GROUP BY p.profile_id
ORDER BY publications DESC
LIMIT 10;
